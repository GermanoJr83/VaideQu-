    <script>

        // Elementos do DOM

        const origemInput = document.getElementById('origem');

        const destinoInput = document.getElementById('destino');

        const compareBtn = document.getElementById('compareBtn');

        const resultsSection = document.getElementById('resultsSection');

        const uberPriceEl = document.getElementById('uberPrice');

        const uberTimeEl = document.getElementById('uberTime');

        const ninetyninePriceEl = document.getElementById('ninetyninePrice');

        const ninetynineTimeEl = document.getElementById('ninetynineTime');

        const savingsSection = document.getElementById('savingsSection');

        const savingsAmountEl = document.getElementById('savingsAmount');

        const addToPiggyBankBtn = document.getElementById('addToPiggyBankBtn');

        const piggyBankBalanceEl = document.getElementById('piggyBankBalance');

 

        let currentSavings = 0;

        let piggyBankTotal = 0;

 

        // Função para formatar moeda

        function formatCurrency(value) {

            return value.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' });

        }

 

        // Evento de clique no botão Comparar

        compareBtn.addEventListener('click', () => {

            const origem = origemInput.value;

            const destino = destinoInput.value;

 

            if (!origem || !destino) {

                alert('Por favor, preencha os campos de origem e destino.');

                return;

            }

 

            // --- SIMULAÇÃO DE DADOS DA API ---

            // Numa aplicação real, aqui seria a chamada às APIs da Uber, 99, etc.

            // Para este esboço, vamos gerar valores aleatórios simples.

 

            const basePrice = 10 + Math.random() * 20; // Preço base entre R$10 e R$30

 

            // Simulação para Uber

            const uberPriceMultiplier = 0.9 + Math.random() * 0.4; // Variação de -10% a +30%

            const uberPrice = basePrice * uberPriceMultiplier;

            const uberTime = 5 + Math.floor(Math.random() * 15); // Tempo entre 5 e 20 min

 

            // Simulação para 99

            const ninetyninePriceMultiplier = 0.85 + Math.random() * 0.5; // Variação de -15% a +35%

            const ninetyninePrice = basePrice * ninetyninePriceMultiplier;

            const ninetynineTime = 7 + Math.floor(Math.random() * 18); // Tempo entre 7 e 25 min

            // --- FIM DA SIMULAÇÃO ---

 

            // Exibir preços e tempos

            uberPriceEl.textContent = formatCurrency(uberPrice);

            uberTimeEl.textContent = `Estimativa: ${uberTime} min`;

            ninetyninePriceEl.textContent = formatCurrency(ninetyninePrice);

            ninetynineTimeEl.textContent = `Estimativa: ${ninetynineTime} min`;

            resultsSection.classList.remove('hidden');

 

            // Calcular e exibir economia

            const cheaperOption = Math.min(uberPrice, ninetyninePrice);

            const moreExpensiveOption = Math.max(uberPrice, ninetyninePrice);

            

            currentSavings = moreExpensiveOption - cheaperOption;

 

            if (currentSavings > 0) {

                savingsAmountEl.textContent = formatCurrency(currentSavings);

                savingsSection.classList.remove('hidden');

            } else {

                savingsAmountEl.textContent = formatCurrency(0);

                savingsSection.classList.add('hidden'); // Esconde se não houver economia

            }

        });

 

        // Evento para adicionar ao cofrinho

        addToPiggyBankBtn.addEventListener('click', () => {

            if (currentSavings > 0) {

                piggyBankTotal += currentSavings;

                piggyBankBalanceEl.textContent = formatCurrency(piggyBankTotal);

               

                // Opcional: Limpar economia atual após adicionar ao cofrinho

                // currentSavings = 0;

                // savingsAmountEl.textContent = formatCurrency(0);

                // savingsSection.classList.add('hidden');

                alert('Economia adicionada ao Cofrinho Inteligente!');

            } else {

                alert('Não há economia para adicionar nesta viagem.');

            }

        });

 

        // Inicializar o cofrinho (poderia ser carregado do localStorage numa versão mais avançada)

        piggyBankBalanceEl.textContent = formatCurrency(piggyBankTotal);

 

    </script>
